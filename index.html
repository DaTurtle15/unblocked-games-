// ============================================================ // app.js — Homepage Logic // ============================================================ const LS_RECENT = 'av_recent'; const LS_FAVORITES = 'av_favorites'; const MAX_RECENT = 8; // ── State ────────────────────────────────────────────────── let activeCategory = 'all'; let searchQuery = ''; // ── Utils ────────────────────────────────────────────────── function getRecent() { return JSON.parse(localStorage.getItem(LS_RECENT) || '[]'); } function getFavorites() { return JSON.parse(localStorage.getItem(LS_FAVORITES) || '[]'); } function showToast(msg) { const t = document.getElementById('toast'); t.textContent = msg; t.classList.add('show'); setTimeout(() => t.classList.remove('show'), 2500); } // ── Category emoji ───────────────────────────────────────── function catEmoji(cat) { const map = { action: '⚡', arcade: '🕹️', puzzle: '🧩', sports: '⚽', multiplayer: '👥' }; return map[cat] || '🎮'; } // ── Build a game card element ────────────────────────────── function buildCard(game, size = 'normal') { const favs = getFavorites(); const isFav = favs.includes(game.id); const card = document.createElement('a'); card.className = 'game-card'; card.href = `game.html?id=${game.id}`; card.setAttribute('data-id', game.id); card.innerHTML = `
${catEmoji(game.category)}
${game.category}
▶
${game.title}
${size !== 'compact' ? `
${game.description}
` : ''}
`; // Favorite toggle — prevent navigation card.querySelector('.card-fav-btn').addEventListener('click', (e) => { e.preventDefault(); e.stopPropagation(); toggleFavorite(game.id); renderAll(); updateFavBadge(); }); return card; } function toggleFavorite(id) { let favs = getFavorites(); if (favs.includes(id)) { favs = favs.filter(f => f !== id); showToast('Removed from favorites'); } else { favs.unshift(id); showToast('❤️ Added to favorites!'); } localStorage.setItem(LS_FAVORITES, JSON.stringify(favs)); } function updateFavBadge() { const favs = getFavorites(); const badge = document.getElementById('favCount'); if (favs.length > 0) { badge.textContent = favs.length; badge.style.display = ''; } else { badge.style.display = 'none'; } } // ── Render categories ────────────────────────────────────── function renderCategories() { const bar = document.getElementById('categoryBar'); bar.innerHTML = ''; CATEGORIES.forEach(cat => { const btn = document.createElement('button'); btn.className = `cat-btn ${cat.id === activeCategory ? 'active' : ''}`; btn.innerHTML = `${cat.icon} ${cat.label}`; btn.addEventListener('click', () => { activeCategory = cat.id; renderAll(); renderCategories(); }); bar.appendChild(btn); }); } // ── Filter games ─────────────────────────────────────────── function getFiltered() { return GAMES.filter(g => activeCategory === 'all' || g.category === activeCategory); } // ── Render all sections ───────────────────────────────────── function renderAll() { const searchActive = searchQuery.trim().length > 0; const mainContent = document.getElementById('mainContent'); const searchResult = document.getElementById('searchResults'); if (searchActive) { mainContent.style.display = 'none'; searchResult.style.display = ''; renderSearch(); return; } mainContent.style.display = ''; searchResult.style.display = 'none'; renderRecent(); renderFavorites(); renderFeatured(); renderAllGames(); } function renderSearch() { const q = searchQuery.trim().toLowerCase(); const found = GAMES.filter(g => g.title.toLowerCase().includes(q) || g.description.toLowerCase().includes(q) || g.category.toLowerCase().includes(q) ); document.getElementById('searchTerm').textContent = `"${searchQuery}"`; document.getElementById('searchCount').textContent = `${found.length} game${found.length !== 1 ? 's' : ''} found`; const grid = document.getElementById('searchGrid'); grid.innerHTML = ''; if (found.length === 0) { grid.innerHTML = `
🔍

No games found for "${searchQuery}"

Try a different search term or browse by category
`; } else { found.forEach(g => grid.appendChild(buildCard(g))); } } function renderRecent() { const recent = getRecent(); const section = document.getElementById('recentSection'); if (recent.length === 0) { section.style.display = 'none'; return; } section.style.display = ''; const grid = document.getElementById('recentGrid'); grid.innerHTML = ''; recent .map(id => GAMES.find(g => g.id === id)) .filter(Boolean) .forEach(g => grid.appendChild(buildCard(g, 'compact'))); } function renderFavorites() { const favs = getFavorites(); const section = document.getElementById('favoritesSection'); if (favs.length === 0) { section.style.display = 'none'; return; } section.style.display = ''; const grid = document.getElementById('favoritesGrid'); grid.innerHTML = ''; favs .map(id => GAMES.find(g => g.id === id)) .filter(Boolean) .forEach(g => grid.appendChild(buildCard(g, 'compact'))); } function renderFeatured() { const games = GAMES.filter(g => g.featured); const grid = document.getElementById('featuredGrid'); grid.innerHTML = ''; games.forEach(g => grid.appendChild(buildCard(g))); } function renderAllGames() { const filtered = getFiltered(); const title = document.getElementById('allGamesTitle'); const count = document.getElementById('gameCount'); const grid = document.getElementById('allGamesGrid'); const catLabel = activeCategory === 'all' ? 'All Games' : (CATEGORIES.find(c => c.id === activeCategory)?.label || 'Games'); title.innerHTML = ` ${catLabel}`; count.textContent = `${filtered.length} game${filtered.length !== 1 ? 's' : ''}`; grid.innerHTML = ''; if (filtered.length === 0) { grid.innerHTML = `
🎮

No games in this category yet

Check back soon — more games are coming!
`; } else { filtered.forEach(g => grid.appendChild(buildCard(g))); } } // ── Search wiring ────────────────────────────────────────── function wireSearch(inputEl, clearEl) { if (!inputEl) return; inputEl.addEventListener('input', () => { searchQuery = inputEl.value; if (clearEl) clearEl.classList.toggle('visible', searchQuery.length > 0); renderAll(); // Sync the other search input const others = document.querySelectorAll('.search-input, .hero-search'); others.forEach(el => { if (el !== inputEl) el.value = searchQuery; }); }); inputEl.addEventListener('keydown', (e) => { if (e.key === 'Escape') { searchQuery = ''; inputEl.value = ''; if (clearEl) clearEl.classList.remove('visible'); renderAll(); } }); if (clearEl) { clearEl.addEventListener('click', () => { searchQuery = ''; inputEl.value = ''; clearEl.classList.remove('visible'); const others = document.querySelectorAll('.search-input, .hero-search'); others.forEach(el => el.value = ''); renderAll(); }); } } // ── Init ─────────────────────────────────────────────────── document.addEventListener('DOMContentLoaded', () => { renderCategories(); renderAll(); updateFavBadge(); wireSearch(document.getElementById('heroSearch'), null); wireSearch(document.getElementById('headerSearch'), document.getElementById('headerSearchClear')); // Scroll to section if hash present if (window.location.hash === '#favorites' || window.location.hash === '#recent') { setTimeout(() => { const el = document.querySelector(window.location.hash); if (el) el.scrollIntoView({ behavior: 'smooth', block: 'start' }); }, 200); } });
